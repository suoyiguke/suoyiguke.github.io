数组的优势和劣势数组的基本知识我懂了，那么，使用数组这种数据结构有什么优势和劣势呢？ 数组拥有非常高效的随机访问能力，只要给出下标，就可以用常量时间找到对应元素。

有一种高效查找元素的算法叫作二分查找，就是利用了数组的这个优势。 至于数组的劣势，体现在插入和删除元素方面。 由于数组元素连续紧密地存储在内存中，插入、删除元素都会导致大量元素被迫移动，影响效率。

总的来说，数组所适合的是读操作多、写操作 少的场景，下一节我们要讲解的链表则恰恰相反。好了，让我们 下一节再会！

###自己的实现

package com.company;

import java.util.Arrays;

class MyArrayList {


    private Object[] array;
    //容量大小
    private int size;

    //真实大小||当前索引
    private int num;

    public MyArrayList(int size) {
        this.size = size;
        this.array = new Object[size];
        this.num = 0;
    }


    /**
     * add
     */
    private Object get(int index) {
        if (index < 0 || index >= size) {
            throw new IndexOutOfBoundsException("超出数组实际元素范围！");
        }
        return array[index];
    }


    /**
     * add
     */
    private void add(Object e) {
        if (null == e) {
            throw new NullPointerException("add 元素为空");
        }

        if (array.length - (1 + num) < 0) {
            resize();
        }

        this.array[num] = e;
        num++;
    }


    /**
     * add index
     */
    private void add(int index, Object e) {
        if (null == e) {
            throw new NullPointerException("add 元素为空");
        }
        if (index < 0 || index >= size) {
            throw new IndexOutOfBoundsException("超出数组实际元素范围！");
        }
        //如果实际元素达到数组容量上限，则对数组进行扩容 20.
        if (size - num <= 0) {
            resize();
        }
        // 需要倒着遍历 右移
        for (int i = this.size-2; i >= index; i--) {
            this.array[i+1] = this.array[i];
        }
        this.num++;
        this.array[index] = e;
    }


    /**
     * resize
     */
    private void resize() {
        Object[] newArray = new Object[this.array.length * 2];
        System.arraycopy(array, 0, newArray, 0, array.length);
        array = newArray;
        size = array.length;
    }

    @Override
    public String toString() {
        return "MyArrayList{" +
                "array=" + Arrays.toString(array) +
                ", size=" + size +
                ", num=" + num +
                '}';
    }

    /**
     * remove(i)
     */
    private void remove(int index) {
        if (index < 0 || index >= size) {
            throw new IndexOutOfBoundsException("超出数组实际元素范围！");
        }

        // 左移，覆盖index的元素
        for (int i = index; i < this.size - 1; i++) {
            this.array[i] = this.array[i + 1];
        }
        this.num--;

    }

    /**
     * remove()
     */
    private void remove() {
        if (num-1 < 0 || num-1 >= size) {
            throw new IndexOutOfBoundsException("超出数组实际元素范围！");
        }
        this.array[num-1] = null;
    }


    public static void main(String[] args) {
        MyArrayList myArrayList = new MyArrayList(3);
        myArrayList.add("yinkai");
        myArrayList.add("yyyy");
        myArrayList.add("zzzz");
        myArrayList.add("ffff");



        myArrayList.remove();
        myArrayList.remove();
        myArrayList.remove();

        myArrayList.add("zzzzzzzzz");
        myArrayList.add(123123);
        myArrayList.add(444444);
        myArrayList.add(6666666);
        myArrayList.add(1);
        myArrayList.add(2);
        myArrayList.add(3);
        myArrayList.add(4);


        System.out.println(myArrayList);


    }


}
